1//! Utilities for interpreting headers from a UDP message.
2
3use crate::gps_time::GpsTime;
4
5/// Marker word for "start of header".
6pub const HEADER_MARKER: &[u8; 4] = &[0xFF, 0xFF, 0xFF, 0xFF];
7
8/// Marker for a "veto frame" data packet.
9const VETO_FRAME_HEADER: &[u8; 4] = &[0xFC, 0xFF, 0xFF, 0xFF];
10
11/// Marker for a "sample environment" data packet.
12const SE_FRAME_HEADER: &[u8; 4] = &[0xFD, 0xFF, 0xFF, 0xFF];
13
14/// Marker for a neutron event data packet.
15const NEUTRON_HEADER: &[u8; 4] = &[0xFF, 0xFF, 0xFF, 0xFF];
16
17/// Length of header in bytes (16 4-byte words).
18pub const HEADER_LEN_BYTES: usize = 16 * 4;
19
20/// View onto a UDP message byte-slice.
21///
22/// This struct provides helper methods for interpreting the bytes from the header of a UDP message.
23pub struct UdpMessageView<'a> {
24    content: &'a [u8],
25}
26
27impl<'a> UdpMessageView<'a> {
28    /// Create a new view onto a UDP message, if the slice is long enough to contain a header,
29    /// starts with a header marker, and is a multiple of 4-byte words.
30    pub fn new(content: &[u8]) -> Option<UdpMessageView<'_>> {
31        (content.len() >= HEADER_LEN_BYTES
32            && content.starts_with(HEADER_MARKER)
33            && content.len().is_multiple_of(4))
34        .then_some(UdpMessageView { content })
35    }
36
37    /// Extract a single word from the header
38    fn header_word(&self, n: usize) -> [u8; 4] {
39        assert!(n <= 15, "Invalid word requested from header");
40        self.content[4 * n..4 * n + 4]
41            .try_into()
42            .expect("slice of length 4")
43    }
44
45    /// Frame number.
46    pub fn frame_number(&self) -> u32 {
47        u32::from_be_bytes(self.header_word(3))
48    }
49
50    /// Total events in this ISIS frame.
51    ///
52    /// Note: this is not the same as the total events in this UDP message; an ISIS
53    /// frame may be split over multiple messages.
54    pub fn events_in_frame(&self) -> u32 {
55        u32::from_be_bytes(self.header_word(7))
56    }
57
58    /// Raw protons-per-pulse per frame; u16 exactly as transmitted over UDP.
59    pub fn raw_ppp_per_frame(&self) -> u16 {
60        u16::from_be_bytes(self.header_word(8)[0..2].try_into().unwrap())
61    }
62
63    /// uAh delivered during this ISIS frame.
64    pub fn ppp_per_frame(&self) -> f64 {
65        const RAW_TO_UAH_SCALING: f64 = 1.738e-6;
66        self.raw_ppp_per_frame() as f64 * RAW_TO_UAH_SCALING
67    }
68
69    /// Veto bits, as transmitted over UDP.
70    pub fn vetoes(&self) -> u16 {
71        u16::from_be_bytes(self.header_word(9)[0..2].try_into().unwrap())
72    }
73
74    /// Period number.
75    pub fn period_number(&self) -> u16 {
76        u16::from_be_bytes(self.header_word(6)[0..2].try_into().unwrap())
77    }
78
79    /// GPS timestamp of this message.
80    pub fn gps_time(&self) -> GpsTime {
81        GpsTime::from_packed_repr(u64::from_be_bytes(
82            self.content[4 * 4..6 * 4]
83                .try_into()
84                .expect("slice of length 8"),
85        ))
86    }
87
88    /// Packet type.
89    pub fn packet_type(&self) -> UdpPacketType {
90        match &self.header_word(1) {
91            NEUTRON_HEADER => UdpPacketType::NeutronData,
92            VETO_FRAME_HEADER => UdpPacketType::VetoFrame,
93            SE_FRAME_HEADER => UdpPacketType::SampleEnvironment,
94            _ => UdpPacketType::Invalid,
95        }
96    }
97
98    /// Get the non-header bytes from this message.
99    ///
100    /// For neutron frames, these bytes contain the neutron event data.
101    pub fn data_bytes(&self) -> &[u8] {
102        &self.content[HEADER_LEN_BYTES..]
103    }
104}
105
106/// Types of packets we may receive over UDP.
107#[derive(Debug, strum::EnumIter, strum::IntoStaticStr)]
108pub enum UdpPacketType {
109    VetoFrame,
110    SampleEnvironment,
111    NeutronData,
112    Invalid,
113}
114
115impl UdpPacketType {
116    pub fn as_prometheus_label(&self) -> &'static str {
117        self.into()
118    }
119}

121#[cfg(test)]
122mod tests {
123    use super::*;
124
125    fn make_raw_udp_message(num_events: usize, ppp: u16, gps_time: u64) -> Vec<u8> {
126        // Note: 4-byte words
127        // Total header length: 64 bytes (16 words)
128        [255_u8; 4] // Header word 0: 'running' header marker
129            .iter()
130            .chain(NEUTRON_HEADER) // Header word 1: neutron data header marker
131            .chain(&[0_u8; 4]) // Header word 2: information
132            .chain(&[0_u8; 4]) // Header word 3: frame number
133            .chain(&gps_time.to_be_bytes()) // Header words 4 & 5: GPS timestamp
134            .chain(&[0_u8; 2]) // Header word 6: period number
135            .chain(&[0_u8; 2]) // Header word 6: unused
136            .chain(&(num_events as u32).to_be_bytes()) // Header word 7: events in frame
137            .chain(&ppp.to_be_bytes()) // Header word 8: ppp_in_frame
138            .chain(&[0_u8; 2]) // Header word 8: unused
139            .chain(&[0_u8; 4]) // Header word 9: vetoes
140            .chain(&[0_u8; 4]) // Header word 10: address of next frame
141            .chain(&[0_u8; 4]) // Header word 11: address of next frame (word address)
142            .chain(&[0_u8; 4]) // Header word 12: streamed frame number
143            .chain(&[0_u8; 4]) // Header word 13: not used
144            .chain(&[0_u8; 4]) // Header word 14: not used
145            .chain(&[0_u8; 4]) // Header word 15: not used
146            .copied()
147            .collect()
148    }
149
150    #[test]
151    fn test_header() {
152        let msg = make_raw_udp_message(10, 23, 0);
153        let header = UdpMessageView::new(&msg).unwrap();
154
155        assert_eq!(header.events_in_frame(), 10);
156        assert_eq!(header.raw_ppp_per_frame(), 23);
157    }
