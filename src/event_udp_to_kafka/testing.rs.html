1//! Testing & benchmarking utilities.
2//!
3//! These utilities are not used at runtime.
4use crate::udp_message::{HEADER_LEN_WORDS, NEUTRON_HEADER};

6/// A valid GPS timestamp
7pub const TESTING_TIMESTAMP: u64 = (26 << (32 + 24))  // 2026
8    + (106 << (32 + 15))  // April 16th
9    + (17 << (32 + 10))  // hour 17
10    + (9 << (32 + 4))  // minute 9
11    + (35 << 30)  // second 35
12    + (123 << 20)  // millisecond 123
13    + (456 << 10)  // microsecond 456
14    + (789); // nanosecond 789
15
16pub const TESTING_TIMESTAMP_NS_SINCE_EPOCH: u64 = 1776359375123456789;
17
18/// Fabricate a valid neutron header.
19pub fn make_raw_neutron_udp_header(num_events: usize, ppp: u8) -> Vec<u8> {
20    let packet_length_words = HEADER_LEN_WORDS + (num_events * 2);
21
22    [0xFF; 4] // Header word 0: 'running' header marker
23        .iter()
24        .chain(NEUTRON_HEADER) // Header word 1: neutron data header marker
25        .chain(&[0_u8; 4]) // Header word 2: information
26        .chain(&[0_u8; 4]) // Header word 3: frame number
27        .chain(&TESTING_TIMESTAMP.to_be_bytes()) // Header words 4 & 5: GPS timestamp
28        .chain(&[0_u8; 2]) // Header word 6: period number
29        .chain(&[0_u8; 2]) // Header word 6: unused
30        .chain(&(num_events as u32).to_be_bytes()) // Header word 7: events in frame
31        .chain(&[ppp]) // Header word 8: ppp_in_frame
32        .chain(&[0_u8; 3]) // Header word 8: unused bits
33        .chain(&[0_u8; 4]) // Header word 9: vetoes
34        .chain(&[0_u8; 4]) // Header word 10: address of next frame
35        .chain(&[0_u8; 4]) // Header word 11: address of next frame (word address)
36        .chain(&[0_u8; 4]) // Header word 12: streamed frame number
37        .chain(&(packet_length_words as u32).to_be_bytes()) // Header word 13: number of 32-bit words in *this* message (header + data)
38        .chain(&[0_u8; 4]) // Header word 14: not used
39        .chain(&[0_u8; 4]) // Header word 15: not used
40        .copied()
41        .collect()
42}
